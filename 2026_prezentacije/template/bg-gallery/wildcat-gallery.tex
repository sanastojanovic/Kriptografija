% Gallery slide helper
%
% \galleryslide{Display Name}{option-string}{pattern command}
%
% Draws the pattern across the full slide, overlays a white header band at
% the top, and shows the display name and the \usetheme option string.
% Each slide is plain (no frame title bar or footer).
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand{\galleryslide}[3]{%
  {%
    \setbeamertemplate{background}{%
      \begin{tikzpicture}
        \useasboundingbox (0,0) rectangle(\the\paperwidth,\the\paperheight);
        #3
        \fill[white] (0,\the\paperheight) rectangle(\the\paperwidth,\the\paperheight);
      \end{tikzpicture}%
    }%
    \setbeamertemplate{headline}{}%
    \setbeamertemplate{footline}{}%
    \begin{frame}[plain]%
      \vskip0.07\paperheight%
      \centering%
      {\color{white}\usebeamerfont{frametitle}#1\par}%
      \vskip0.025\paperheight%
      {\color{white}\small\texttt{\textbackslash usetheme[pattern=#2]\{wildcat\}}\par}%
    \end{frame}%
  }%
}
